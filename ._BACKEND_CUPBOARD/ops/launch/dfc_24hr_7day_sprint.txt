4. STRIPE CONNECT ONBOARDING LINK (Node.js pseudocode):
   const accountLink = await stripe.accountLinks.create({
     account: connectedAccountId,
     refresh_url: 'https://datafightcentral.web.app/onboard/refresh',
     return_url: 'https://datafightcentral.web.app/onboard/complete',
     type: 'account_onboarding'
   });

5. TOKEN ENDPOINT SNIPPET (Node.js + jsonwebtoken):
   const jwt = require('jsonwebtoken');
   app.post('/api/entitlement/:id/token', authMiddleware, async (req, res) => {
     const entitlement = await db.getEntitlement(req.params.id);
     if (!entitlement || entitlement.userId !== req.user.id)
       return res.status(403).send();
     const payload = {
       iss: 'dfc-backend',
       sub: `user:${req.user.id}`,
       ent: `ent:${entitlement.id}`,
       evt: `event:${entitlement.eventId}`,
       dev: req.body.deviceFingerprint || 'unknown',
       iat: Math.floor(Date.now()/1000),
       exp: Math.floor(Date.now()/1000) + 600,
       jti: uuidv4(),
       aud: 'cdn.datafightcentral.com'
     };
     const token = jwt.sign(payload, process.env.SIGNING_PRIVATE_KEY, { algorithm: 'RS256' });
     const manifestUrl = `https://cdn.datafightcentral.com/event/${entitlement.eventId}/playlist.m3u8?token=${encodeURIComponent(token)}`;
     await db.saveToken(payload.jti, entitlement.id, req.user.id);
     res.json({ playbackToken: token, manifestUrl });
   });
